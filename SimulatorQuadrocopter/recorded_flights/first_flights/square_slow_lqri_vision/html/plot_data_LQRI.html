close all
%1: time
%2: x
%3: x_dot
%4: pitch
%5: x_ref
%6: pitch_cmd
%7: y
%8: y_dot
%9: roll
%10: y_ref
%11: roll_cmd
%12: z
%13: z_dot
%14: z_ref
%15: thrust cmd
%16: yaw
%17: ang vel z
%18: yaw ref
%19: yaw rate cmd

load hquad_traj_lqri.mat


start_time=1;
end_time=1400;

% get trajectory data
hquad_traj_lqri=hquad_traj_lqri(:,start_time:end_time);

% move starting point to 0/0
hquad_traj_lqri(2,:)=hquad_traj_lqri(2,:)-1;
hquad_traj_lqri(5,:)=hquad_traj_lqri(5,:)-1;
hquad_traj_lqri(7,:)=hquad_traj_lqri(7,:)+0.4;
hquad_traj_lqri(10,:)=hquad_traj_lqri(10,:)+0.4;
hquad_traj_lqri(12,:)=hquad_traj_lqri(12,:)+1;
hquad_traj_lqri(14,:)=hquad_traj_lqri(14,:)+1;

% data is recorded with z-axis pointing down !!
% change this
hquad_traj_lqri(7:19,:)=-1*hquad_traj_lqri(7:19,:);

x - axis

figure
plot(hquad_traj_lqri(1,:)'-hquad_traj_lqri(1,1),[hquad_traj_lqri(2,:)',...
    hquad_traj_lqri(3,:)',hquad_traj_lqri(5,:)']);
title('Trajectory Control with LQRI');
legend('position','velocity','reference','Location','SouthEast');
ylabel('position (meter)');
xlabel('time (sec)');
axis([0 25 -1.2 1.2]);
